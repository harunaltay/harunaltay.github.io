(2022 06 10) Mod_7_100_e_kadar_Solution (iş sıralamasına dikkat)

/*

1 den 100 e kadar olan 7 ye bölünebilen sayıların
- listesi
- toplamı
- ortalaması

+ 1 den 100 e kadar olan sayıları KONSOLA sırala, alt alta
+ bunların içinde... 7 ye bölünebilenleri KONSOLA sırala
+ bu 7 ye bölünebilenleri bir de adetlerini bul!
+ bunları topla
double ortalama = int toplam / int adet

*/
int kacTaneBulundu = 0;
int toplam = 0;
double ortalama;

for (int i = 1; i <= 100; i++)
{
int kalan = i % 7;
if (kalan == 0)
{
kacTaneBulundu++;
toplam = toplam + i;
Console.WriteLine(kacTaneBulundu + " : " + i);
}
}

Console.WriteLine("Toplam: " + toplam);

ortalama = Convert.ToDouble(toplam) / Convert.ToDouble(kacTaneBulundu);
Console.WriteLine("ortalama: " + ortalama);

double saglama = ortalama * kacTaneBulundu;
Console.WriteLine("sağlama: " + saglama);

Console.ReadKey();
